import React, { useEffect, useRef } from "react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Terminal, Copy, Download } from "lucide-react";
import { Button } from "@/components/ui/button";

export default function LogViewer({ logs }) {
  const logEndRef = useRef(null);

  useEffect(() => {
    logEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [logs]);

  const copyLogs = () => {
    navigator.clipboard.writeText(logs || "");
  };

  return (
    <Card className="glass border-white/10 shadow-2xl overflow-hidden">
      <CardHeader className="border-b border-white/10 bg-black/40">
        <div className="flex items-center justify-between">
          <CardTitle className="flex items-center gap-3 text-white text-xl font-black">
            <div className="w-10 h-10 bg-gradient-to-br from-green-500 to-emerald-600 rounded-xl flex items-center justify-center shadow-lg">
              <Terminal className="w-5 h-5 text-white" />
            </div>
            LOGS EM TEMPO REAL
          </CardTitle>
          <Button
            variant="outline"
            size="sm"
            onClick={copyLogs}
            className="glass border-white/20 hover:bg-white/10 font-semibold"
            disabled={!logs}
          >
            <Copy className="w-4 h-4 mr-2" />
            Copiar
          </Button>
        </div>
      </CardHeader>
      <CardContent className="p-0">
        <div className="bg-black p-6 font-mono text-sm max-h-96 overflow-y-auto">
          {logs ? (
            <pre className="text-green-400 whitespace-pre-wrap leading-relaxed">{logs}</pre>
          ) : (
            <div className="text-center py-12">
              <Terminal className="w-12 h-12 text-gray-700 mx-auto mb-4" />
              <p className="text-gray-600 italic font-medium">
                Aguardando atividade do bot...
              </p>
            </div>
          )}
          <div ref={logEndRef} />
        </div>
      </CardContent>
    </Card>
  );
}
